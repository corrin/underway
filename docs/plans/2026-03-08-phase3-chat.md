async def execute_tool(
    tool_name: str, arguments: dict[str, Any], user_id: uuid.UUID, session: AsyncSession
) -> dict[str, Any]:
    """Dispatch a tool call to the appropriate async handler."""
    handler = _TOOL_HANDLERS.get(tool_name)
    if not handler:
        return {"error": f"Unknown tool: {tool_name}"}
    return await handler(arguments, user_id, session)


async def _get_tasks(
    arguments: dict[str, Any], user_id: uuid.UUID, session: AsyncSession
) -> dict[str, Any]:
    stmt = select(Task).where(Task.user_id == user_id)
    status = arguments.get("status")
    if status:
        stmt = stmt.where(Task.status == status)
    priority = arguments.get("priority")
    if priority is not None:
        stmt = stmt.where(Task.priority == priority)
    stmt = stmt.order_by(Task.position)
    result = await session.execute(stmt)
    tasks = list(result.scalars().all())
    return {"tasks": [t.to_dict() for t in tasks]}


async def _complete_task(
    arguments: dict[str, Any], user_id: uuid.UUID, session: AsyncSession
) -> dict[str, Any]:
    task_id_str = arguments.get("task_id", "")
    try:
        task_uuid = uuid.UUID(task_id_str)
    except (ValueError, AttributeError):
        return {"success": False, "error": f"Invalid task ID: {task_id_str}"}
    result = await session.execute(select(Task).where(Task.id == task_uuid, Task.user_id == user_id))
    task = result.scalar_one_or_none()
    if not task:
        return {"success": False, "error": "Task not found."}
    task.status = "completed"
    await session.flush()
    return {"success": True, "task": task.to_dict()}


async def _create_task(
    arguments: dict[str, Any], user_id: uuid.UUID, session: AsyncSession
) -> dict[str, Any]:
    title = arguments.get("title", "")
    description = arguments.get("description")
    priority = arguments.get("priority")
    content_hash = hashlib.sha256(
        json.dumps({"title": title, "description": description, "priority": priority}, sort_keys=True).encode()
    ).hexdigest()
    task = Task(
        user_id=user_id,
        title=title,
        description=description,
        priority=priority,
        status="active",
        provider="local",
        provider_task_id=str(uuid.uuid4()),
        content_hash=content_hash,
        list_type="unprioritized",
        position=0,
    )
    session.add(task)
    await session.flush()
    return {"success": True, "task": task.to_dict()}


async def _update_task(
    arguments: dict[str, Any], user_id: uuid.UUID, session: AsyncSession
) -> dict[str, Any]:
    task_id_str = arguments.get("task_id", "")
    try:
        task_uuid = uuid.UUID(task_id_str)
    except (ValueError, AttributeError):
        return {"success": False, "error": f"Invalid task ID: {task_id_str}"}
    result = await session.execute(select(Task).where(Task.id == task_uuid, Task.user_id == user_id))
    task = result.scalar_one_or_none()
    if not task:
        return {"success": False, "error": "Task not found."}
    for field in ("title", "description", "priority", "status"):
        if field in arguments:
            setattr(task, field, arguments[field])
    await session.flush()
    return {"success": True, "task": task.to_dict()}

from __future__ import annotations

from typing import ClassVar

from fastrest.fields import CharField, DateTimeField, IntegerField, ListField, UUIDField
from fastrest.serializers import ModelSerializer, Serializer

from aligned.models.conversation import ChatMessage, Conversation


class ConversationSerializer(ModelSerializer):
    id = UUIDField(read_only=True)
    user_id = UUIDField(read_only=True)
    title = CharField(required=False, allow_null=True)
    created_at = DateTimeField(read_only=True)
    updated_at = DateTimeField(read_only=True)

    class Meta:
        model = Conversation
        fields: ClassVar[list[str]] = ["id", "user_id", "title", "created_at", "updated_at"]
        read_only_fields: ClassVar[list[str]] = ["id", "user_id", "created_at", "updated_at"]


class ChatMessageSerializer(ModelSerializer):
    id = UUIDField(read_only=True)
    conversation_id = UUIDField(read_only=True)
    role = CharField(read_only=True)
    content = CharField(read_only=True, allow_null=True)
    tool_calls = ListField(required=False, read_only=True)
    tool_call_id = CharField(required=False, read_only=True, allow_null=True)
    sequence = IntegerField(read_only=True)
    created_at = DateTimeField(read_only=True)

    class Meta:
        model = ChatMessage
        fields: ClassVar[list[str]] = [
            "id", "conversation_id", "role", "content",
            "tool_calls", "tool_call_id", "sequence", "created_at",
        ]
        read_only_fields: ClassVar[list[str]] = [
            "id", "conversation_id", "role", "content",
            "tool_calls", "tool_call_id", "sequence", "created_at",
        ]


class ChatInputSerializer(Serializer):
    message = CharField(required=True)
    conversation_id = UUIDField(required=False)

"""Conversation viewset — CRUD + message history."""

from __future__ import annotations

from typing import TYPE_CHECKING

from fastrest.decorators import action
from fastrest.viewsets import ModelViewSet
from sqlalchemy import select

from aligned.models.conversation import ChatMessage, Conversation
from aligned.serializers.chat import ChatMessageSerializer, ConversationSerializer
from aligned.viewsets.base import SessionMixin

if TYPE_CHECKING:
    from fastrest.request import Request


class ConversationViewSet(SessionMixin, ModelViewSet):
    queryset = Conversation
    serializer_class = ConversationSerializer
    lookup_field_type = str

    async def get_queryset(self) -> list[Conversation]:
        """Scope to current user, ordered by most recent."""
        user = self.request.user
        session = self._session
        stmt = (
            select(Conversation)
            .where(Conversation.user_id == user.id)
            .order_by(Conversation.updated_at.desc())
        )
        result = await session.execute(stmt)
        return list(result.scalars().all())

    async def perform_create(self, serializer: ConversationSerializer) -> None:
        """Set user_id when creating a conversation."""
        user = self.request.user
        session = self._session
        data = dict(serializer.validated_data)
        conv = Conversation(user_id=user.id, **data)
        session.add(conv)
        await session.flush()
        serializer.instance = conv

    @action(methods=["get"], detail=True, url_path="messages")
    async def messages(self, request: Request, pk: str = "", **kwargs: str) -> list[dict[str, object]]:
        """GET /api/conversations/{pk}/messages — full message history."""
        from fastapi import HTTPException

        user = request.user
        session = self._session
        import uuid as _uuid

        result = await session.execute(
            select(Conversation).where(
                Conversation.id == _uuid.UUID(pk), Conversation.user_id == user.id
            )
        )

"""Chat streaming endpoint with SSE and agentic tool-calling loop."""

from __future__ import annotations

import json
import uuid
from collections.abc import AsyncGenerator
from typing import Any

import litellm
from fastapi import APIRouter, HTTPException, Request
from fastapi.responses import StreamingResponse
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from aligned.chat.tools import MUTATING_TOOLS, TOOL_DEFINITIONS, execute_tool
from aligned.models.conversation import ChatMessage, Conversation
from aligned.models.task import Task
from aligned.models.user import User

router = APIRouter()

SYSTEM_PROMPT = (
    "You are Aligned, a task management assistant. You help users:\n"
    "- Break down complex tasks into smaller, actionable subtasks\n"
    "- Track progress by marking tasks complete\n"
    "- Prioritize and reorder their task list\n"
    "- Plan their day using their calendar and tasks\n\n"
    "Be concise and action-oriented. When a user mentions completing something, "
    "use the complete_task tool. When they want to add something, use create_task. "
    "Proactively suggest breaking down large tasks."
)


async def _get_user(user_id: uuid.UUID, session: AsyncSession) -> User | None:
    """Load full user from DB (needed for api_key, instructions, model)."""
    result = await session.execute(select(User).where(User.id == user_id))
    return result.scalar_one_or_none()


def _build_messages(
    history: list[ChatMessage], user_message: str, ai_instructions: str | None
) -> list[dict[str, Any]]:
    """Build messages list: system prompt + history + new user message."""
    system_content = SYSTEM_PROMPT
    if ai_instructions:
        system_content += f"\n\nAdditional instructions: {ai_instructions}"
    messages: list[dict[str, Any]] = [{"role": "system", "content": system_content}]
    for msg in history:
        messages.append(msg.to_dict())
    messages.append({"role": "user", "content": user_message})
    return messages


async def _save_message(
    session: AsyncSession,
    conversation_id: uuid.UUID,
    role: str,
    content: str | None,
    sequence: int,
    tool_calls: Any | None = None,
    tool_call_id: str | None = None,
) -> None:
    """Save a single message to the database."""
    msg = ChatMessage(
        conversation_id=conversation_id,
        role=role,
        content=content,
        tool_calls=tool_calls,
        tool_call_id=tool_call_id,
        sequence=sequence,
    )
    session.add(msg)
    await session.flush()


async def _generate_sse(
    conversation: Conversation,
    messages: list[dict[str, Any]],
    model: str,
    api_key: str,
    session: AsyncSession,
    user_id: uuid.UUID,
) -> AsyncGenerator[str, None]:
    """Async generator yielding SSE events with tool-calling loop."""
    current_messages = list(messages)
    next_sequence = len(conversation.messages) + 1  # +1 because user msg already saved

    try:
        while True:
            response = await litellm.acompletion(
                model=model,
                api_key=api_key,
                messages=current_messages,
                tools=TOOL_DEFINITIONS,
                stream=True,
            )

            collected_content = ""
            collected_tool_calls: dict[int, dict[str, Any]] = {}

            async for chunk in response:
                delta = chunk.choices[0].delta

                if delta.content:
                    collected_content += delta.content
                    yield f"data: {json.dumps({'type': 'token', 'content': delta.content})}\n\n"

                if delta.tool_calls:
                    for tc in delta.tool_calls:
                        idx: int = tc.index
                        if idx not in collected_tool_calls:
                            collected_tool_calls[idx] = {
                                "id": "",
                                "function": {"name": "", "arguments": ""},
                            }
                        if tc.id:
                            collected_tool_calls[idx]["id"] = tc.id
                        if tc.function:
                            if tc.function.name:
                                collected_tool_calls[idx]["function"]["name"] = tc.function.name
                            if tc.function.arguments:
                                collected_tool_calls[idx]["function"]["arguments"] += tc.function.arguments

            if collected_tool_calls:
                tool_calls_list = [collected_tool_calls[i] for i in sorted(collected_tool_calls)]

                # Save assistant message with tool_calls
                tc_data = [
                    {"id": tc["id"], "type": "function", "function": tc["function"]}
                    for tc in tool_calls_list
                ]
                await _save_message(
                    session, conversation.id, "assistant", "", next_sequence, tool_calls=tc_data
                )
                next_sequence += 1

                # Execute tools and save results
                any_mutating = False
                tool_results: list[dict[str, Any]] = []
                for tc in tool_calls_list:
                    func = tc["function"]
                    name: str = func["name"]
                    try:
                        arguments: dict[str, Any] = json.loads(func["arguments"])
                    except (json.JSONDecodeError, TypeError):
                        arguments = {}

                    result = await execute_tool(name, arguments, user_id, session)
                    tc_id: str = tc["id"]
                    tool_results.append({"tool_call_id": tc_id, "name": name, "result": result})

                    # Save tool result message
                    await _save_message(
                        session, conversation.id, "tool", json.dumps(result),
                        next_sequence, tool_call_id=tc_id,
                    )
                    next_sequence += 1

                    yield f"data: {json.dumps({'type': 'tool_call', 'name': name, 'result': result})}\n\n"
                    if name in MUTATING_TOOLS:
                        any_mutating = True

                if any_mutating:
                    yield f"data: {json.dumps({'type': 'dashboard_refresh'})}\n\n"

                # Add to messages for next LLM iteration
                current_messages.append(
                    {"role": "assistant", "content": "", "tool_calls": tc_data}
                )
                for tr in tool_results:
                    current_messages.append(
                        {
                            "role": "tool",
                            "content": json.dumps(tr["result"]),
                            "tool_call_id": tr["tool_call_id"],
                        }
                    )
                continue

            # No tool calls — final response
            await _save_message(session, conversation.id, "assistant", collected_content, next_sequence)
            yield f"data: {json.dumps({'type': 'done', 'conversation_id': str(conversation.id)})}\n\n"
            break

    except Exception as e:
        yield f"data: {json.dumps({'type': 'error', 'message': str(e)})}\n\n"


@router.post("/api/chat")
async def chat(request: Request) -> StreamingResponse:
    """Main chat endpoint. Returns SSE stream."""
    session: AsyncSession = request.state.db_session

    # Authenticate
    auth_user = request.state.user
    if not auth_user:
        raise HTTPException(status_code=401, detail="Not authenticated")

    user = await _get_user(auth_user.id, session)
    if not user:
        raise HTTPException(status_code=401, detail="User not found")

    data: dict[str, Any] = await request.json()
    message = (data.get("message") or "").strip()
    if not message:
        raise HTTPException(status_code=400, detail="Message is required")

    if not user.ai_api_key:
        raise HTTPException(status_code=400, detail="No API key configured. Go to Settings to add one.")

    # Get or create conversation
    conversation_id_str = data.get("conversation_id")
    if conversation_id_str:
        result = await session.execute(
            select(Conversation).where(
                Conversation.id == uuid.UUID(conversation_id_str),
                Conversation.user_id == user.id,
            )
        )
        conversation = result.scalar_one_or_none()
        if not conversation:
            raise HTTPException(status_code=404, detail="Conversation not found")
    else:
        conversation = Conversation(user_id=user.id, title=message[:100])
        session.add(conversation)
        await session.flush()

    # Save user message
    next_seq = len(conversation.messages)
    await _save_message(session, conversation.id, "user", message, next_seq)

    # Build messages from history (includes the message we just saved after flush)
    await session.refresh(conversation)
    model = user.llm_model or "gpt-4o"
    llm_messages = _build_messages(conversation.messages[:-1], message, user.ai_instructions)

    return StreamingResponse(
        _generate_sse(conversation, llm_messages, model, user.ai_api_key, session, user.id),
        media_type="text/event-stream",
        headers={"Cache-Control": "no-cache", "X-Accel-Buffering": "no"},
    )

export async function sendMessage(
  message: string,
  conversationId?: string,
  onEvent?: (event: SSEEvent) => void,
): Promise<void> {
  const token = localStorage.getItem('token')
  const response = await fetch('/api/chat', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Accept: 'text/event-stream',
      Authorization: `Bearer ${token}`,
    },
    body: JSON.stringify({ message, conversation_id: conversationId }),
  })

  if (!response.ok) {
    const error = await response.json()
    throw new Error(error.detail || 'Chat request failed')
  }

  const reader = response.body?.getReader()
  if (!reader) return

  const decoder = new TextDecoder()
  let buffer = ''

  while (true) {
    const { done, value } = await reader.read()
    if (done) break

    buffer += decoder.decode(value, { stream: true })
    const lines = buffer.split('\n')
    buffer = lines.pop() ?? ''

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        try {
          const event: SSEEvent = JSON.parse(line.slice(6))
          onEvent?.(event)
        } catch {
          // Skip malformed events
        }
      }
    }
  }
}

### Task 6: Chat Pinia Store

**Files:**
- Create: `frontend/src/stores/chat.ts`

**Step 1: Create the store**

Create `frontend/src/stores/chat.ts`:

```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import {
  fetchConversations,
  createConversation,
  deleteConversation,
  fetchMessages,
  fetchDashboard,
  sendMessage,
  type Conversation,
  type ChatMessage,
  type DashboardData,
  type SSEEvent,
} from '@/api/chat'

export const useChatStore = defineStore('chat', () => {
  const conversations = ref<Conversation[]>([])
  const currentConversation = ref<Conversation | null>(null)
  const messages = ref<ChatMessage[]>([])
  const streamingContent = ref('')
  const isStreaming = ref(false)
  const error = ref('')
  const dashboard = ref<DashboardData | null>(null)

  async function loadConversations(): Promise<void> {
    try {
      conversations.value = await fetchConversations()
    } catch (e) {
      error.value = e instanceof Error ? e.message : 'Failed to load conversations'
    }
  }

  async function loadMessages(conversationId: string): Promise<void> {
    try {
      messages.value = await fetchMessages(conversationId)
    } catch (e) {
      error.value = e instanceof Error ? e.message : 'Failed to load messages'
    }
  }

  async function selectConversation(conv: Conversation): Promise<void> {
    currentConversation.value = conv
    await loadMessages(conv.id)
  }

  async function startNewConversation(): Promise<void> {
    currentConversation.value = null
    messages.value = []
    streamingContent.value = ''
  }

  async function removeConversation(id: string): Promise<void> {
    try {
      await deleteConversation(id)
      conversations.value = conversations.value.filter((c) => c.id !== id)
      if (currentConversation.value?.id === id) {
        currentConversation.value = null
        messages.value = []
      }
    } catch (e) {
      error.value = e instanceof Error ? e.message : 'Failed to delete conversation'
    }
  }

  async function send(messageText: string): Promise<void> {
    error.value = ''
    isStreaming.value = true
    streamingContent.value = ''

    // Add user message to local state immediately
    const userMsg: ChatMessage = {
      id: crypto.randomUUID(),
      conversation_id: currentConversation.value?.id ?? '',
      role: 'user',
      content: messageText,
      tool_calls: null,
      tool_call_id: null,
      sequence: messages.value.length,
      created_at: new Date().toISOString(),
    }
    messages.value.push(userMsg)

    try {
      await sendMessage(messageText, currentConversation.value?.id ?? undefined, (event: SSEEvent) => {
        switch (event.type) {
          case 'token':
            streamingContent.value += event.content ?? ''
            break
          case 'tool_call':
            // Could show tool calls in UI
            break
          case 'dashboard_refresh':
            loadDashboard()
            break
          case 'done':
            if (event.conversation_id && !currentConversation.value) {
              // New conversation created — set it and reload list
              currentConversation.value = {
                id: event.conversation_id,
                user_id: '',
                title: messageText.slice(0, 100),
                created_at: new Date().toISOString(),
                updated_at: new Date().toISOString(),
              }
              loadConversations()
            }
            // Add assistant message to local state
            messages.value.push({
              id: crypto.randomUUID(),
              conversation_id: currentConversation.value?.id ?? '',
              role: 'assistant',
              content: streamingContent.value,
              tool_calls: null,
              tool_call_id: null,
              sequence: messages.value.length,
              created_at: new Date().toISOString(),
            })
            streamingContent.value = ''
            break
          case 'error':
            error.value = event.message ?? 'An error occurred'
            break
        }
      })
    } catch (e) {
      error.value = e instanceof Error ? e.message : 'Failed to send message'
    } finally {
      isStreaming.value = false
    }
  }

  async function loadDashboard(): Promise<void> {
    try {
      dashboard.value = await fetchDashboard()
    } catch {
      // Dashboard is non-critical
    }
  }

  return {
    conversations,
    currentConversation,
    messages,
    streamingContent,
    isStreaming,
    error,
    dashboard,
    loadConversations,
    loadMessages,
    selectConversation,
    startNewConversation,
    removeConversation,
    send,
    loadDashboard,
  }
})
```

  const text = messageInput.value.trim()

  await store.send(text)
  await nextTick()
  scrollToBottom()

  }
